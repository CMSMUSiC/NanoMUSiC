371  // Clear `TriggerBits` and save the seed trigger.
372  // Will loop over the `TriggerBits`. Once a fired trigger is found:
373  // 1 - the fired trigger bit is saved as trigger_seed
374  // 2 - all others bits are set to false
376  EventData &set_trigger_bits()
377  {
378  if (*this)
379  {
380  switch (year)
381  {
382  case Year::Run2016APV:
383  break;
384  case Year::Run2016:
385  break;
386  case Year::Run2017:
387  trigger_bits.set("SingleMuonLowPt", event_info.HLT_IsoMu27.get())
388  .set("SingleMuonHighPt",
389  event_info.HLT_Mu50.get() || event_info.HLT_TkMu100.get() || event_info.HLT_OldMu100.get())
390  .set("SingleElectron", false)
391  .set("DoubleMuon", false)
392  .set("DoubleElectron", false)
393  .set("Photon", false)
394  .set("Tau", false)
395  .set("BJet", false)
396  .set("Jet", false)
397  .set("MET", false);
398  break;
399  case Year::Run2018:
400  break;
401  default:
402  throw std::runtime_error("Year (" + std::to_string(year) +
403  ") not matching with any possible Run2 cases (2016APV, 2016, 2017 or 2018).");
404  }

412  bool found_hana_key = false;

414  if (HANA_FIRST_TO_SV(stream) == trigger_stream)

416  found_hana_key = true;
417  boost::hana::for_each( //

419  trigger_bits.set(trigger_path, false); //
420  });
421  }
422  });
423 
424  // just a sanitiy check ...
425  if (not found_hana_key)
426  {
427  throw std::runtime_error(
428  fmt::format("[ ERROR ] The requested Trigger Stream ({}) could not be found.", trigger_stream));
