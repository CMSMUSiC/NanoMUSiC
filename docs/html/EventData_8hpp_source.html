338  .set(TriggerBits::HLTPath.index_of("DoubleMuon"), false)

344 
345  switch (year)
346  {
347  case Year::Run2016APV:
348  break;
349  case Year::Run2016:
350  break;
351  case Year::Run2017:
352  trigger_bits.set(TriggerBits::HLTPath.index_of("SingleMuonLowPt"), event_info.HLT_IsoMu27.get())
353  .set(TriggerBits::HLTPath.index_of("SingleMuonHighPt"),
354  event_info.HLT_Mu50.get() || event_info.HLT_TkMu100.get() || event_info.HLT_OldMu100.get())
355  .set(TriggerBits::HLTPath.index_of("SingleElectron"), false)
356  .set(TriggerBits::HLTPath.index_of("DoubleMuon"), false)
357  .set(TriggerBits::HLTPath.index_of("DoubleElectron"), false)
358  .set(TriggerBits::HLTPath.index_of("Tau"), false)
359  .set(TriggerBits::HLTPath.index_of("BJet"), false)
360  .set(TriggerBits::HLTPath.index_of("MET"), false)
361  .set(TriggerBits::HLTPath.index_of("Photon"), false);
362  break;
363  case Year::Run2018:
364  break;
365  default:
366  throw std::runtime_error("Year (" + std::to_string(year) +
367  ") not matching with any possible Run2 cases (2016APV, 2016, 2017 or 2018).");
368  }
369

371  if (trigger_bits.any())
372  {
