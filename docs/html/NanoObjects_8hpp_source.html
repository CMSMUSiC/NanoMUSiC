18 inline auto mass(const float &pt1,
19  const float &eta1,
20  const float &phi1,
21  const float &mass1,
22  const float &pt2,
23  const float &eta2,
24  const float &phi2,
25  const float &mass2) -> float
26 {
27  // Conversion from (pt, eta, phi, mass) to (x, y, z, e) coordinate system
28  const auto x1 = pt1 * std::cos(phi1);
29  const auto y1 = pt1 * std::sin(phi1);
30  const auto z1 = pt1 * std::sinh(eta1);
31  const auto e1 = std::sqrt(x1 * x1 + y1 * y1 + z1 * z1 + mass1 * mass1);
32 
33  const auto x2 = pt2 * std::cos(phi2);
34  const auto y2 = pt2 * std::sin(phi2);
35  const auto z2 = pt2 * std::sinh(eta2);
36  const auto e2 = std::sqrt(x2 * x2 + y2 * y2 + z2 * z2 + mass2 * mass2);
37 
38  // Addition of particle four-vector elements
39  const auto e = e1 + e2;
40  const auto x = x1 + x2;
41  const auto y = y1 + y2;
42  const auto z = z1 + z2;
43 
44  return std::sqrt(e * e - x * x - y * y - z * z);
45 }

47 inline auto pt(const float &pt1, const float &phi1, const float &pt2, const float &phi2) -> float
48 {
49  const auto x1 = pt1 * std::cos(phi1);
50  const auto y1 = pt1 * std::sin(phi1);
51 
52  const auto x2 = pt2 * std::cos(phi2);
53  const auto y2 = pt2 * std::sin(phi2);
54 
55  // Addition of particle 2d-vector components
56  const auto x = x1 + x2;
57  const auto y = y1 + y2;
58 
59  return std::sqrt(x * x + y * y);
60 }
