126 void splitString(std::vector<T> &result, const std::string &input, const std::string &sep = ",",
127  bool ignoreEmpty = false)
128 {
129  result.clear();
130  if (ignoreEmpty && input.empty())
131  return;
132 
133  std::string::size_type pos = input.find(sep);
134  std::string::size_type pos_old = 0;
135 
136  size_t sep_len = sep.size();
137 
138  while (std::string::npos != pos)
139  {
140  size_t len = pos - pos_old;
141 
142  if (len != 0)
143  {
144  std::string substring = input.substr(pos_old, len);
145  substring = boost::trim_copy(substring);
146  T temp = T();
147  if (!substring.empty())
148  temp = fromString<T>(substring);
149  result.push_back(temp);
150  }
151  else if (!ignoreEmpty)
152  {
153  result.push_back(T());
154  }
155 
156  pos_old = pos + sep_len;
157 
158  pos = input.find(sep, pos_old);
159  }
160 
161  if (pos_old != input.size())
162  {
163  std::string substring = input.substr(pos_old);
164  substring = boost::trim_copy(substring);
165  T temp = T();
166  if (!substring.empty())

169  }
