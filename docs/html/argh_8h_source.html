195  inline void parser::parse(int argc, const char* const argv[], int mode /*= PREFER_FLAG_FOR_UNREG_OPTION*/)
196  {
197  // clear out possible previous parsing remnants
198  flags_.clear();
199  params_.clear();
200  pos_args_.clear();
201 
202  // convert to strings
203  args_.resize(static_cast<decltype(args_)::size_type>(argc));
204  std::transform(argv, argv + argc, args_.begin(), [](const char* const arg) { return arg; });
205 
206  // parse line
207  for (auto i = 0u; i < args_.size(); ++i)
208  {
209  if (!is_option(args_[i]))
210  {
211  pos_args_.emplace_back(args_[i]);
212  continue;
213  }
214 
215  auto name = trim_leading_dashes(args_[i]);
216 
217  if (!(mode & NO_SPLIT_ON_EQUALSIGN))
218  {
219  auto equalPos = name.find('=');
220  if (equalPos != std::string::npos)
221  {
222  params_.insert({ name.substr(0, equalPos), name.substr(equalPos + 1) });
223  continue;
224  }
225  }
226 
227  // if the option is unregistered and should be a multi-flag
228  if (1 == (args_[i].size() - name.size()) && // single dash
229  argh::parser::SINGLE_DASH_IS_MULTIFLAG & mode && // multi-flag mode
230  !is_param(name)) // unregistered
231  {
232  std::string keep_param;
233 
234  if (!name.empty() && is_param(std::string(1ul, name.back()))) // last char is param
235  {
236  keep_param += name.back();
237  name.resize(name.size() - 1);
238  }
239 
240  for (auto const& c : name)
241  {
242  flags_.emplace(std::string{ c });
243  }
244 
245  if (!keep_param.empty())
246  {
247  name = keep_param;
248  }
249  else
250  {
251  continue; // do not consider other options for this arg
252  }
253  }
254 
255  // any potential option will get as its value the next arg, unless that arg is an option too
256  // in that case it will be determined a flag.
257  if (i == args_.size() - 1 || is_option(args_[i + 1]))
258  {
259  flags_.emplace(name);
260  continue;
261  }
262 
263  // if 'name' is a pre-registered option, then the next arg cannot be a free parameter to it is skipped
264  // otherwise we have 2 modes:
265  // PREFER_FLAG_FOR_UNREG_OPTION: a non-registered 'name' is determined a flag.
266  // The following value (the next arg) will be a free parameter.
267  //
268  // PREFER_PARAM_FOR_UNREG_OPTION: a non-registered 'name' is determined a parameter, the next arg
269  // will be the value of that option.
270 
271  assert(!(mode & argh::parser::PREFER_FLAG_FOR_UNREG_OPTION)
272  || !(mode & argh::parser::PREFER_PARAM_FOR_UNREG_OPTION));
273 
274  bool preferParam = mode & argh::parser::PREFER_PARAM_FOR_UNREG_OPTION;
275 
276  if (is_param(name) || preferParam)
277  {
278  params_.insert({ name, args_[i + 1] });
279  ++i; // skip next value, it is not a free parameter
280  continue;
281  }
282  else
283  {
284  flags_.emplace(name);
285  }
286  }
287  }
