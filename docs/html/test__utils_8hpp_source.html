85 struct check_equal_fn
86 {
87  CPP_template(typename T, typename U)(
88  requires(!both_ranges<T, U>)) //
89  constexpr void operator()(
90  T && actual, U && expected,
91  source_location sloc = source_location::current()) const
92  {
93  CHECK_SLOC(sloc, (T &&) actual == (U &&) expected);
94  }
95 
96  CPP_template(typename Rng1, typename Rng2)(
97  requires both_ranges<Rng1, Rng2>)
98  constexpr void operator()(
99  Rng1 && actual, Rng2 && expected,
100  source_location sloc = source_location::current()) const
101  {
102  auto begin0 = ranges::begin(actual);
103  auto end0 = ranges::end(actual);
104  auto begin1 = ranges::begin(expected);
105  auto end1 = ranges::end(expected);
106  for(; begin0 != end0 && begin1 != end1; ++begin0, ++begin1)
107  (*this)(*begin0, *begin1, sloc);
108  CHECK_SLOC(sloc, begin0 == end0);
109  CHECK_SLOC(sloc, begin1 == end1);
110  }

114  constexpr void operator()(
115  Rng && actual, std::initializer_list<Val> && expected,
116  source_location sloc = source_location::current()) const
117  {
118  (*this)(actual, expected, sloc);
