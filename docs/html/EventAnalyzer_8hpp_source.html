202  template <typename T, typename F>
203  auto get_self_cleanning_mask(const T objects, RVec<int> mask, float max_delta_r, F &&filter) -> RVec<int>
204  {
205  auto cleanning_mask = RVec<int>(objects.size, 1);
206 
207  if (objects.size >= 2)
208  {
209  for (std::size_t i = 0; i < objects.size; i++)
210  {
211  if (mask[i])
212  {
213  auto cleanning_result = true;
214  for (std::size_t j = 0; j < objects.eta.size(); j++)
215  {
216  if (i != j)
217  {
218  float delta_phi = VecOps::DeltaPhi(objects.phi[i], objects.phi[i]);
219  float delta_eta = std::abs(objects.eta[i] - objects.eta[i]);
220  float delta_r = std::sqrt(delta_phi * delta_phi + delta_eta * delta_eta);
221  if (delta_r < max_delta_r)
222  {
223  cleanning_result = cleanning_result && filter(objects, i, j);
224  }
225  if (cleanning_result == false)
226  {
227  break;
228  }
229  }
230  }
231  cleanning_mask[i] = static_cast<int>(cleanning_result);
232  }
233  }
234  }
235  return cleanning_mask;
236  }

243  auto get_cross_cleanning_mask(const T1 &obj1, const T2 &obj2, const RVec<int> &mask, float max_delta_r) -> RVec<int>
244  {
245  auto cleanning_mask = RVec<int>(obj1.eta.size(), 1);
246 
247  if (obj1.eta.size() >= 1 and obj2.eta.size() >= 1)

254  auto delta_eta = VecOps::abs(obj1.eta[i] - obj2.eta);
255  auto delta_r = VecOps::sqrt(delta_phi * delta_phi + delta_eta * delta_eta);
256  cleanning_mask[i] = static_cast<int>(not(VecOps::Any(delta_r < max_delta_r)));
