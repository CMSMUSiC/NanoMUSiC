43 template <typename T1, typename T2>
44 std::pair<RVec<float>, RVec<float>> get_matches(const T1 &trigger_objects_pt, //
45  const T1 &trigger_objects_eta, //
46  const T1 &trigger_objects_phi, //
47  const T2 &nano_objects_pt, //
48  const T2 &nano_objects_eta, //
49  const T2 &nano_objects_phi)
50 {
51  auto matches_distances = RVec<float>(nano_objects_pt.size(), std::numeric_limits<float>::max());
52  auto matches_relative_pT = RVec<float>(nano_objects_pt.size(), std::numeric_limits<float>::max());
53 
54  for (std::size_t idx = 0; idx < nano_objects_pt.size(); idx++)
55  {
56  // DeltaR
57  matches_distances.at(idx) =
58  VecOps::Min(VecOps::sqrt(VecOps::pow((trigger_objects_eta - nano_objects_eta[idx]), 2.) +
59  +VecOps::pow(VecOps::DeltaPhi(trigger_objects_phi, nano_objects_phi[idx]), 2.)));
60 
61  // Relative pT diff
62  matches_relative_pT.at(idx) =
63  VecOps::Min(VecOps::abs(nano_objects_pt[idx] - trigger_objects_phi) / (nano_objects_pt[idx]));
64  }
65 
66  return std::make_pair(matches_distances, matches_relative_pT);
67 }
