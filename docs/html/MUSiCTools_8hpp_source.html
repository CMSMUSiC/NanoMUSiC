228 template <class T>
229 void splitString(std::vector<T> &result, const std::string &input, const std::string &sep = ",", bool ignoreEmpty = false)
230 {
231  result.clear();
232  if (ignoreEmpty && input.empty())
233  return;
234 
235  std::string::size_type pos = input.find(sep);
236  std::string::size_type pos_old = 0;
237 
238  size_t sep_len = sep.size();
239 
240  while (std::string::npos != pos)
241  {
242  size_t len = pos - pos_old;
243 
244  if (len != 0)
245  {
246  std::string substring = input.substr(pos_old, len);
247  substring = boost::trim_copy(substring);
248  T temp = T();
249  if (!substring.empty())
250  temp = fromString<T>(substring);
251  result.push_back(temp);
252  }
253  else if (!ignoreEmpty)
254  {
255  result.push_back(T());
256  }
257 
258  pos_old = pos + sep_len;
259 
260  pos = input.find(sep, pos_old);
261  }
262 
263  if (pos_old != input.size())
264  {
265  std::string substring = input.substr(pos_old);
266  substring = boost::trim_copy(substring);
267  T temp = T();
268  if (!substring.empty())
269  temp = fromString<T>(substring);
270  result.push_back(temp);
271  }
272  else if (!ignoreEmpty)
273  {
274  result.push_back(T());
275  }
276 }
