126  void ImportFile(const std::string &iPath, const std::string &old_string = "", const std::string &new_string = "")
127  {
128  Path importPath = boost::trim_copy(Tools::ExpandPath(iPath));
129  Path absPath = Init(importPath);
130  std::ifstream configFile(absPath.string().c_str());
131 
132  if (!configFile.is_open())
133  throw std::runtime_error("MConfig::ImportFile(): Failed to open Config File: '" + absPath.string() + "'");
134 
135  unsigned int lineNum = 0;
136  while (configFile.good() && !configFile.eof())
137  {
138  std::string line;
139  getline(configFile, line);
140  line = Tools::removeComment(line, m_commentChar);
141  ++lineNum;
142 
143  ScanForImports(line, lineNum, absPath);
144 
145  if (!line.empty())
146  {
147  // search for the first appearance of delimiterChar and quoteChar
148  std::string::size_type foundDelim = line.find_first_of(m_delimiterChar);
149  std::string::size_type foundQuote = line.find_first_of(m_quoteChar);
150 
151  // if there is no delimiter found in the line, something is wrong
152  if (foundDelim == std::string::npos)
153  {
154  std::stringstream error;
155  error << "MConfig::ImportFile( const std::string &path ): No delimiter '" << m_delimiterChar
156  << "' found on line number " << lineNum << " in file '" << absPath << "'. Please check it!";
157  throw std::runtime_error(error.str());
158 
159  // it is allowed to have a delimiterChar inside the quotation, e.g.: a = "b = c"
160  // but there must be a delimiter before the first quoteChar
161  }
162  else if (foundQuote < foundDelim)
163  {
164  std::stringstream error;
165  error << "MConfig::ImportFile( const std::string &path ): Quotation character '" << m_quoteChar
166  << "' found before delimiter '" << m_delimiterChar << "' on line number " << lineNum
167  << " in file '" << absPath << "'. Please check it!";
168  throw std::runtime_error(error.str());
169  }
170 
171  std::string key(line.begin(), line.begin() + foundDelim);
172  std::string value = line.substr(foundDelim + 1);
173 
174  // now search for delimiterChar and quoteChar in the value string
175  foundDelim = value.find_first_of(m_delimiterChar);
176  foundQuote = value.find_first_of(m_quoteChar);
177 
178  // if the delimiter is not inside the quotation now, something is wrong
179  if (foundDelim < foundQuote)
180  {
181  std::stringstream error;
182  error << "MConfig::ImportFile( const std::string &path ): More than one delmiter '"
183  << m_delimiterChar << "' found on line number " << lineNum << " in file '" << absPath
184  << "'. Please check it!";
185  throw std::runtime_error(error.str());
186  }
187 
188  boost::trim(key);
189  boost::trim(value);
190  value = ExtractFromQuotes(value, lineNum, absPath);
191 
192  if (old_string != "" and key.find(old_string) != std::string::npos)
193  {
194  size_t pos = key.find(old_string);
195  key.replace(pos, old_string.length(), new_string);
196  }
197  AddItem(key, value);
198  }
199  }

339  // take 'line' and extract the (sub)string between the quotation characters
340  // everything beyond the quotation is ignored!
341  std::string ExtractFromQuotes(const std::string &line, const unsigned int lineNum, const Path &filePath)
342  {
343  std::string::size_type posFirst = line.find_first_of(m_quoteChar);
344  std::string::size_type posLast = line.find_last_of(m_quoteChar);
345 
346  // if there are no quotations, fine
347  if ((posFirst == std::string::npos) && (posLast == std::string::npos))
348  return line;
349 
350  // if there is only one quotatation, not so nice
351  if (posFirst == posLast)
352  {
353  std::stringstream error;
354  error << "MConfig::ExtractFromQuotes( std::string line ): Only one quotation character '" << m_quoteChar
355  << "' found on line: " << lineNum << " in file: '" << filePath << "'. Please check it!";
356  throw std::runtime_error(error.str());
357  }
358 
359  std::string result(line.begin() + ++posFirst, line.begin() + posLast);
360 
361  // if there are more than two quotation characters, that's bad
362  if (result.find(m_quoteChar) != std::string::npos)
363  {
364  std::stringstream error;
365  error << "MConfig::ExtractFromQuotes( std::string line ): Too many quotation characters on line num: '"
366  << lineNum << "', in file: '" << filePath << "'. Only two quotation characters allowed per line!";
367  throw std::runtime_error(error.str());
368  }
369 
370  return result;
371  }

412  void ScanForImports(std::string &line, const unsigned int lineNum, const Path &filePath)
413  {
414  // Format: import "file" with "OLD_STRING" as "NEW_STRING"
415  if (line.empty() or line.find(m_importIdentifier) == std::string::npos)
416  return;
417 
418  // line = boost::trim_copy( line ); This removes only trailing and leading white spaces.
419  // Better:
420  line.erase(std::unique(line.begin(), line.end(), [](char a, char b) { return a == ' ' && b == ' '; }),
421  line.end());
422 
423  std::vector<std::string> substrings = Tools::splitString<std::string>(line, false, " ");
424 
425  // check import
426  if (substrings[0] == m_importIdentifier)
427  {
428  std::string importPath = ExtractFromQuotes(substrings[1], lineNum, filePath);
429  if (substrings.size() == 2)
430  {
431  line.clear();
432  ImportFile(importPath);
433  }
434  else // expect name modifier
435  {
436  std::string new_string = "";
437  std::string old_string = "";
438  if (substrings.size() != 6)
439  {
440  std::stringstream error;
441  error << "MConfig::ScanForImports( std::string line, const unsigned int lineNum, const Path "
442  "&filePath ): Expected"
443  << "2 or 6 words but got " << substrings.size() << " on line: " << lineNum << " in file: '"
444  << filePath << "'. Please check it!";
445  throw std::runtime_error(error.str());
446  }
447  if (substrings[2] != m_modifiedImportIdentifier1)
448  {
449  std::stringstream error;
450  error << "MConfig::ScanForImports( std::string line, const unsigned int lineNum, const Path "
451  "&filePath ): Expected '"
452  << m_modifiedImportIdentifier1 << "' but found '" << substrings[2] << "' on line: " << lineNum
453  << " in file: '" << filePath << "'. Please check it!";
454  throw std::runtime_error(error.str());
455  }
456  if (substrings[4] != m_modifiedImportIdentifier2)
457  {
458  std::stringstream error;
459  error << "MConfig::ScanForImports( std::string line, const unsigned int lineNum, const Path "
460  "&filePath ): Expected '"
461  << m_modifiedImportIdentifier2 << "' but found '" << substrings[4] << "' on line: " << lineNum
462  << " in file: '" << filePath << "'. Please check it!";
463  throw std::runtime_error(error.str());
464  }
465  new_string = ExtractFromQuotes(substrings[5], lineNum, filePath);
466  old_string = ExtractFromQuotes(substrings[3], lineNum, filePath);
467  line.clear();
468  ImportFile(importPath, old_string, new_string);
469  }
470  }
