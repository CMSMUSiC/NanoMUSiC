328  auto operator()(float eta_sc, float pt, std::string_view variation = "nominal") const -> double
329  {
330  // check for physical values
331  if (pt < 0.)
332  {
333  throw std::runtime_error("[ Electron Trigger SF ] The value of pT should be positive.");
334  }
335 
336  // check if is barrel or endcap
337  if (std::fabs(eta_sc) < 1.442)
338  {
339  // check limits
340  if (pt >= barrel_pt_upper_limit)
341  {
342  pt = barrel_pt_upper_limit - epsilon;
343  }
344  if (pt <= barrel_pt_lower_limit)
345  {
346  pt = barrel_pt_lower_limit + epsilon;
347  }
348  auto nominal = sf_histogram_barrel.GetBinContent(sf_histogram_barrel.FindFixBin(eta_sc, pt));
349  auto uncert = 0.;
350 
351  if (variation == "up")
352  {
353  uncert = sf_histogram_barrel.GetBinErrorUp(sf_histogram_barrel.FindFixBin(eta_sc, pt));
354  }
355  else if (variation == "down")
356  {
357  uncert = -1.0 * sf_histogram_barrel.GetBinErrorLow(sf_histogram_barrel.FindFixBin(eta_sc, pt));
358  }
359 
360  return nominal + uncert;
361  }
362  else if (std::fabs(eta_sc) >= 1.566 and std::fabs(eta_sc) < 2.5)
363  {
364  // check limits
365  if (pt >= endcap_pt_upper_limit)
366  {
367  pt = endcap_pt_upper_limit - epsilon;
368  }
369  if (pt <= endcap_pt_lower_limit)
370  {
371  pt = endcap_pt_lower_limit + epsilon;
372  }
373 
374  auto nominal = sf_histogram_endcap.GetBinContent(sf_histogram_endcap.FindFixBin(eta_sc, pt));
375  auto uncert = 0.;
376 
377  if (variation == "up")
378  {
379  uncert = sf_histogram_barrel.GetBinErrorUp(sf_histogram_barrel.FindFixBin(eta_sc, pt));
380  }
381  else if (variation == "down")
382  {
383  uncert = -1.0 * sf_histogram_barrel.GetBinErrorLow(sf_histogram_barrel.FindFixBin(eta_sc, pt));
384  }
385 
386  return nominal + uncert;
387  }
388  else
389  {
390  throw std::runtime_error(fmt::format(
391  "[ Electron Trigger SF ] The provided Super Cluster eta value ({}), is about of range.", eta_sc));
392  }
393  }
