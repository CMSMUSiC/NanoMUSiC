273  template <typename T, typename BinOp = std::multiplies<T>>
274  auto operator()(const Year &year,
275  const RVec<T> &pt,
276  const RVec<T> &eta,
277  const std::string &variation,
278  BinOp Op = std::multiplies<T>(),
279  float initial_value = 1.f) const -> float
280  {
281  static_assert(std::is_floating_point<T>::value, "The type must be floating point.");
282 
283  if (not is_data)
284  {
285  if (correction_type == "MuonReco"sv //
286  or correction_type == "MuonIdLowPt"sv //
287  or correction_type == "MuonIdHighPt"sv //
288  or correction_type == "MuonIsoLowPt"sv //
289  or correction_type == "MuonIsoHighPt"sv)
290  {
291 
292  RVec<float> sfs = RVec<float>(pt.size(), initial_value);
293 
294  sfs = VecOps::Map(pt, eta, [&](const T &_pt, const T &_eta) {
295  return (*this)({CorrectionHelpers::get_year_for_muon_sf(year), //
296  static_cast<double>(_eta), //
297  static_cast<double>(_pt), //
298  variation});
299  });
300 
301  return std::reduce(sfs.cbegin(), sfs.cend(), initial_value, Op);
302 
303  // auto weight = std::reduce(sfs.cbegin(), sfs.cend(), initial_value, Op);
304  // fmt::print("Weight: {}\n", weight);
305  // return weight;
306  }
307 
308  // default case
309  throw std::runtime_error(
310  fmt::format("No matching was fouund for this correction type: {}.", correction_type));
311  }
312  return 1.;
313  }
