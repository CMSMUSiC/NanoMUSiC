325  auto operator()(float eta_sc, float pt, std::string_view variation = "nominal") const -> double
326  {
327  // check for physical values
328  if (pt < 0.)
329  {
330  throw std::runtime_error("[ Electron Trigger SF ] The value of pT should be positive.");
331  }
332 
333  // check if is barrel or endcap
334  if (std::fabs(eta_sc) < 1.442)
335  {
336  // check limits
337  if (pt >= barrel_pt_upper_limit)
338  {
339  pt = barrel_pt_upper_limit - epsilon;
340  }
341  if (pt <= barrel_pt_lower_limit)
342  {
343  pt = barrel_pt_lower_limit + epsilon;
344  }
345  auto nominal = sf_histogram_barrel.GetBinContent(sf_histogram_barrel.FindFixBin(eta_sc, pt));
346  auto uncert = 0.;
347 
348  if (variation == "up")
349  {
350  uncert = sf_histogram_barrel.GetBinErrorUp(sf_histogram_barrel.FindFixBin(eta_sc, pt));
351  }
352  if (variation == "down")
353  {
354  uncert = -1.0 * sf_histogram_barrel.GetBinErrorLow(sf_histogram_barrel.FindFixBin(eta_sc, pt));
355  }
356 
357  return nominal + uncert;
358  }
359  if (std::fabs(eta_sc) >= 1.566 and std::fabs(eta_sc) < 2.5)
360  {
361  // check limits
362  if (pt >= endcap_pt_upper_limit)
363  {
364  pt = endcap_pt_upper_limit - epsilon;
365  }
366  if (pt <= endcap_pt_lower_limit)
367  {
368  pt = endcap_pt_lower_limit + epsilon;
369  }
370 
371  auto nominal = sf_histogram_endcap.GetBinContent(sf_histogram_endcap.FindFixBin(eta_sc, pt));
372  auto uncert = 0.;
373 
374  if (variation == "up")
375  {
376  uncert = sf_histogram_barrel.GetBinErrorUp(sf_histogram_barrel.FindFixBin(eta_sc, pt));
377  }
378  if (variation == "down")
379  {
380  uncert = -1.0 * sf_histogram_barrel.GetBinErrorLow(sf_histogram_barrel.FindFixBin(eta_sc, pt));
381  }
382 
383  return nominal + uncert;
384  }
385  throw std::runtime_error(fmt::format(
386  "[ Electron Trigger SF ] The provided Super Cluster eta value ({}), is about of range.", eta_sc));
387  }
