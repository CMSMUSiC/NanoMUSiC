103  Corrector(const CorrectionTypes _correction_type, const Year _year, bool _is_data)
104  : correction_type(_correction_type), year(_year), is_data(_is_data)
105  {
106 
107  switch (_correction_type)
108  {
109  case CorrectionTypes::MuonLowPt: {
110  auto [input_file, _dummy_key] = correction_keys.at({correction_type, year});
111  correction_ref = RoccoR(input_file);
112  break;
113  }
114  default:
115  if (!is_data)
116  {
117  auto [json_file, key] = correction_keys.at({correction_type, year});
118  correction_ref = correction::CorrectionSet::from_file(json_file)->at(key);
119  }
120  }
121  }

123  // correctionlib
124  template <class T = const std::vector<std::variant<int, double, std::string>>>
125  double operator()(T &&values)
126  {
127  if (not is_data)
128  {
129  return std::get<CorrectionlibRef_t>(correction_ref)->evaluate(std::forward<T>(values));
130  }
131  return 1.;
132  }
133 
134  // rochester corrections - Data
135  double operator()(const int Q, const double pt, const double eta, const double phi, const int s = 0,
136  const int m = 0)
137  {
138  if (is_data)
139  {
140  return std::get<RochesterCorrection_t>(correction_ref).kScaleDT(Q, pt, eta, phi, s, m);
141  }
142  throw std::runtime_error("The signature of the used method is only valid for Data samples.");
143  }
144 
145  // rochester corrections - MC (matched GenMuon - recommended)
146  double operator()(const int Q, const double pt, const double eta, const double phi, const double genPt,
147  const int s = 0, const int m = 0)
148  {
149  if (not is_data)
150  {
151  return std::get<RochesterCorrection_t>(correction_ref).kSpreadMC(Q, pt, eta, phi, genPt, s, m);
152  }
153  throw std::runtime_error("The signature of the used method is only valid for MC samples.");
154  }
155 
156  // rochester corrections - MC (unmatched GenMuon - not recommended)
157  double operator()(const int Q, const double pt, const double eta, const double phi, const int n, const double u,
158  const int s = 0, const int m = 0)
159  {
160  if (not is_data)

162  return std::get<RochesterCorrection_t>(correction_ref).kSmearMC(Q, pt, eta, phi, n, u, s, m);
163  }
164  throw std::runtime_error("The signature of the used method is only valid for MC samples.");
165  }
