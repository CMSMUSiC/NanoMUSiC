63  Corrector(const CorrectionTypes _correction_type, const Year _year, bool _isData)
64  : correction_type(_correction_type), year(_year), isData(_isData)
65  {
66 
67  switch (_correction_type)
68  {
69  case CorrectionTypes::MuonLowPt: {
70  auto [input_file, _dummy_key] = correction_keys.at({correction_type, year});
71  correction_ref = RoccoR(input_file);
72  break;
73  }
74  default:
75  if (!isData)
76  {
77  auto [json_file, key] = correction_keys.at({correction_type, year});
78  correction_ref = correction::CorrectionSet::from_file(json_file)->at(key);
79  }
80  }
81  }
