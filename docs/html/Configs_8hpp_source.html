37 auto get_runyear(const std::string &year_str)
38 {
39  // check year
40  if (year_str != "2016APV" && year_str != "2016" && year_str != "2017" && year_str != "2018")
41  {
42  auto error_msg = "ERROR: year should be 2016APV, 2016, 2017 or 2018";
43  throw std::out_of_range{error_msg};
44  }
45 
46  // return year as enum
47  if (year_str == "2016APV")
48  {
49  return Year::Run2016APV;
50  }
51  else if (year_str == "2016")
52  {
53  return Year::Run2016;
54  }
55  else if (year_str == "2017")
56  {
57  return Year::Run2017;
58  }
59  else
60  {
61  return Year::Run2018;
62  }
