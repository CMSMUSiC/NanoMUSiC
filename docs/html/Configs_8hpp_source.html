265 // get data_stream from process
266 inline std::string_view get_trigger_stream(bool is_data, const std::string &process)
267 {
268  using namespace std::literals;
269  if (not is_data)
270  {
271  return "MC"sv;
272  }
273  else
274  {
275  std::string_view _buffer = "";
276  boost::hana::for_each(TriggerStreamRedList, [&](const auto &stream) {
277  if (process.rfind(HANA_FIRST_TO_SV(stream), 0) == 0 and _buffer == "")
278  {
279  _buffer = HANA_FIRST_TO_SV(stream);
280  }
281  });
282 
283  if (_buffer != "")
284  {
285  return _buffer;
286  }
287  }
288  throw std::runtime_error("[ ERROR ] Data Stream not found for this process name: \""s + process + "\"."s);
289 }
