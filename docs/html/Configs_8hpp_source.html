91 inline auto get_mass_by_id(const int &id) -> float

108  if (std::abs(id) == MuonNeutrino::Id)
109  {
110  return MuonNeutrino::Mass;
111  }
112 
113  if (std::abs(id) == Tau::Id)
114  {
115  return Tau::Mass;

117 
118  if (std::abs(id) == TauNeutrino::Id)
119  {
120  return TauNeutrino::Mass;
121  }
122 
123  if (std::abs(id) == Gluon::Id)
124  {
125  return Gluon::Mass;
126  }
127 
128  if (std::abs(id) == Photon::Id)

130  return Photon::Mass;
131  }

134  {
135  return Z::Mass;

140  return W::Mass;

143  throw(std::runtime_error(fmt::format("Invalid PDG Id ({}).", id)));
